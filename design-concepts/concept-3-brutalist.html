CAMPAIGN CONTRACT — PSEUDOCODE EXAMPLE
// CITEBACK CAMPAIGN CONTRACT v2.1
// Once deployed, this cannot be modified.
// Not by you. Not by us. Not by anyone.

contract Campaign {

  address   owner     = msg.sender;
  bytes32   contentHash;
  uint256   executeAt;
  uint256   nodeThreshold = 51000; // 55% of network

  // IMMUTABLE after deployment
  function deploy(content, timestamp) {
    contentHash = hash(content);
    executeAt   = timestamp;
    // No cancel function exists.
    // No pause function exists.
    // No override function exists.
  }

  // Triggered by network consensus
  function execute() {
    require(block.timestamp >= executeAt);
    require(nodes.confirmed >= nodeThreshold);
    broadcast(contentHash);
    // Done. Irreversible.
  }

  // This function does not exist:
  // function shutdown() { ... }
}
